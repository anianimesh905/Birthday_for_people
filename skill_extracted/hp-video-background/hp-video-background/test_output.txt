/**
 * initVideoBackground(cfg)
 *
 * Loads the correct Hogwarts house video on page start, then crossfades
 * to a new video whenever the houseChanged event fires.
 *
 * Falls back to an animated gradient if no video file is found or the
 * file fails to load — the site always looks beautiful, video or not.
 *
 * @param {object} cfg - window.siteContent (from content.js)
 */
function initVideoBackground(cfg = {}) {
  const video = document.getElementById('bg-video');
  if (!video) return;

  // Set up the fade-out transition duration (fade-in is set dynamically)
  video.style.transition = 'opacity 0.4s ease';

  // Map each house name to its video file path.
  // Paths are relative to the HTML file — adjust the 'videos/' prefix if needed.
  const VIDEO_MAP = {
    gryffindor: cfg.gryffindorVideo ? `videos/${cfg.gryffindorVideo}` : null,
    slytherin:  cfg.slytherinVideo  ? `videos/${cfg.slytherinVideo}`  : null,
    ravenclaw:  cfg.ravenclawVideo  ? `videos/${cfg.ravenclawVideo}`  : null,
    hufflepuff: cfg.hufflepuffVideo ? `videos/${cfg.hufflepuffVideo}` : null,
  };

  // Track the current house to skip redundant reloads
  let currentHouse = null;

  /**
   * Apply the animated gradient fallback for a given house.
   * Hides the <video> element and shows the gradient on the container instead.
   */
  function applyFallback(houseName) {
    video.style.opacity = '0'; // hide broken video element

    // Target the video container — adjust this selector to match your site's HTML
    const container =
      document.getElementById('bg-video-container') ||
      document.getElementById('video-container')    ||
      document.querySelector('.video-bg-wrap')      ||
      document.body;

    container.classList.remove(
      'fallback-gryffindor',
      'fallback-slytherin',
      'fallback-ravenclaw',
      'fallback-hufflepuff'
    );
    container.classList.add(`fallback-${houseName}`);
  }

  /**
   * Load and play the video for a given house.
   * @param {string}  houseName - one of: gryffindor | slytherin | ravenclaw | hufflepuff
   * @param {boolean} animate   - crossfade (true) or instant load (false for page load)
   */
  function loadHouseVideo(houseName, animate = true) {
    if (houseName === currentHouse) return; // already showing this house
    currentHouse = houseName;

    const src = VIDEO_MAP[houseName];

    // Clear any active gradient fallback — we're attempting a real video
    const container =
      document.getElementById('bg-video-container') ||
      document.getElementById('video-container')    ||
      document.querySelector('.video-bg-wrap')      ||
      document.body;

    container.classList.remove(
      'fallback-gryffindor',
      'fallback-slytherin',
      'fallback-ravenclaw',
      'fallback-hufflepuff'
    );

    // Reassign onerror for this specific load (avoids stacking multiple listeners)
    video.onerror = () => applyFallback(houseName);

    // No video file configured for this house — jump straight to fallback
    if (!src) {
      applyFallback(houseName);
      return;
    }

    if (animate) {
      // ── Step 1: Fade current video out over 400ms ──────────────
      video.style.transition = 'opacity 0.4s ease';
      video.style.opacity = '0';

      setTimeout(() => {
        // ── Step 2: Swap the source ────────────────────────────
        video.src = src;
        video.load();

        // ── Step 3: Fade new video in over 600ms ──────────────
        video.style.transition = 'opacity 0.6s ease';
        video.style.opacity = '1';

        video.play().catch(() => applyFallback(houseName));

        // Restore faster fade-out transition for next switch
        video.addEventListener(
          'canplay',
          () => { video.style.transition = 'opacity 0.4s ease'; },
          { once: true }
        );
      }, 400); // matches the 0.4s fade-out duration

    } else {
      // ── Initial page load: no animation, just play ─────────────
      video.src = src;
      video.load();
      video.style.opacity = '1';
      video.play().catch(() => applyFallback(houseName));
    }
  }

  // ── Listen for house selection events (from hp-house-selector-ui) ──
  document.addEventListener('houseChanged', (e) => {
    const house = (e.detail && e.detail.house) ? e.detail.house : 'gryffindor';
    loadHouseVideo(house, true);
  });

  // ── Initial load on page open ───────────────────────────────────
  const defaultHouse = (cfg.defaultHouse || 'gryffindor').toLowerCase();
  loadHouseVideo(defaultHouse, false);
}

// ── Bootstrap ──────────────────────────────────────────────────────
document.addEventListener('DOMContentLoaded', () => {
  initVideoBackground(window.siteContent || {});
});

/*
═══════════════════════════════════════════════════════════════════
Integration Notes
═══════════════════════════════════════════════════════════════════

JavaScript
  1. Find the getHours() / switch(true) block in your script.js and DELETE it.
  2. Paste initVideoBackground() in its place.
  3. In your DOMContentLoaded handler, the bootstrap at the bottom of this
     file calls it automatically — you don't need to call it twice.
  4. Load order: content.js → house-selector.js → video-background.js

CSS
  Paste the CSS sections into your main stylesheet, or save as video-bg.css:
    <link rel="stylesheet" href="video-bg.css">
  Add before </head>.

Container ID
  The JS looks for #bg-video-container, #video-container, and .video-bg-wrap
  in that order. If your site uses a different element, update the selectors
  in both applyFallback() and loadHouseVideo() in the JS, AND in the CSS block.

Testing fallbacks without videos
  To test the gradient fallbacks, temporarily rename a video file (or set
  the filename in content.js to something that doesn't exist). The onerror
  handler will fire and show the animated gradient. Every house should produce
  a visibly different colour wash.

Video doesn't autoplay?
  Most browsers block autoplay with sound. Make sure #bg-video has:
    <video id="bg-video" autoplay muted loop playsinline>
  The muted and playsinline attributes are required for autoplay to work
  on mobile (especially iOS Safari).
*/
