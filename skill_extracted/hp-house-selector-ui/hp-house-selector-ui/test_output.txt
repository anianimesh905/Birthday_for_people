/**
 * Hogwarts House Selector
 * Requires: siteContent to be available as window.siteContent
 * Dispatches: CustomEvent 'houseChanged' with { house, primary, accent }
 */

const HOUSES = {
  gryffindor: {
    emoji: '🦁',
    label: 'Gryffindor',
    primary: '#740001',
    accent: '#D3A625',
    toastBg: 'rgba(116,0,1,0.92)',
    toastColor: '#ffffff',
    videoSrc: null,
  },
  slytherin: {
    emoji: '🐍',
    label: 'Slytherin',
    primary: '#1A472A',
    accent: '#AAAAAA',
    toastBg: 'rgba(26,71,42,0.92)',
    toastColor: '#ffffff',
    videoSrc: null,
  },
  ravenclaw: {
    emoji: '🦅',
    label: 'Ravenclaw',
    primary: '#0E1A40',
    accent: '#946B2D',
    toastBg: 'rgba(14,26,64,0.92)',
    toastColor: '#ffffff',
    videoSrc: null,
  },
  hufflepuff: {
    emoji: '🦡',
    label: 'Hufflepuff',
    primary: '#ECB939',
    accent: '#2A2A2A',
    toastBg: 'rgba(236,185,57,0.95)',
    toastColor: '#1a1a1a', // dark text — gold bg is too light for white
    videoSrc: null,
  },
};

let toastTimer = null;

/**
 * Show a 2-second slide-up toast with house branding.
 */
function showHouseToast(house) {
  const toast = document.getElementById('house-toast');
  if (!toast) return;

  toast.textContent = `The Sorting Hat has chosen… ${house.label}! ${house.emoji}`;
  toast.style.background = house.toastBg;
  toast.style.color = house.toastColor;
  toast.style.borderColor = house.accent + '66'; // ~40% opacity border

  clearTimeout(toastTimer);
  toast.classList.remove('visible');
  void toast.offsetWidth; // force reflow to restart transition
  toast.classList.add('visible');

  toastTimer = setTimeout(() => toast.classList.remove('visible'), 2000);
}

/**
 * Core house switching logic.
 * @param {string} houseName  - one of: gryffindor | slytherin | ravenclaw | hufflepuff
 * @param {object} options
 * @param {boolean} options.showToast - show the Sorting Hat toast (default true)
 * @param {boolean} options.spin      - spin the badge emoji (default true)
 */
function selectHouse(houseName, { showToast = true, spin = true } = {}) {
  const house = HOUSES[houseName];
  if (!house) {
    console.warn(`[hp-house-selector] Unknown house: "${houseName}"`);
    return;
  }

  // 1. Update active class on all badges
  document.querySelectorAll('.house-badge').forEach(btn => {
    btn.classList.toggle('active', btn.dataset.house === houseName);
  });

  // 2. Spin the selected badge emoji
  if (spin) {
    const emojiEl = document.querySelector(
      `.house-badge[data-house="${houseName}"] .badge-emoji`
    );
    if (emojiEl) {
      emojiEl.classList.remove('spinning');
      void emojiEl.offsetWidth; // reflow — restart animation if re-selecting same house
      emojiEl.classList.add('spinning');
      emojiEl.addEventListener(
        'animationend',
        () => emojiEl.classList.remove('spinning'),
        { once: true }
      );
    }
  }

  // 3. Crossfade background video
  const video = document.getElementById('bg-video');
  if (video && house.videoSrc) {
    video.style.opacity = '0';
    setTimeout(() => {
      video.src = house.videoSrc;
      video.load();
      video.play().catch(() => {
        // Silently ignore autoplay policy blocks — fallback color handles it
      });
      video.style.opacity = '1';
    }, 300); // matches CSS opacity transition duration
  }

  // 4. Update envelope seal color
  const seal = document.getElementById('env-seal');
  if (seal) seal.style.backgroundColor = house.primary;

  // 5. Swap body theme class
  const bodyClasses = Array.from(document.body.classList).filter(
    c => !c.startsWith('theme-')
  );
  document.body.className = bodyClasses.join(' ');
  document.body.classList.add(`theme-${houseName}`);

  // 6. Toast notification
  if (showToast) showHouseToast(house);

  // 7. Notify particle system (and any other listeners)
  document.dispatchEvent(
    new CustomEvent('houseChanged', {
      bubbles: true,
      detail: {
        house: houseName,
        primary: house.primary,
        accent: house.accent,
      },
    })
  );
}

/**
 * Initialise the selector.
 * Reads videoSrc paths from siteContent config, wires click listeners,
 * and activates the defaultHouse silently on page load.
 */
function initHouseSelector(cfg = {}) {
  // Hydrate video sources from siteContent
  HOUSES.gryffindor.videoSrc = cfg.gryffindorVideo ? `videos/${cfg.gryffindorVideo}` : null;
  HOUSES.slytherin.videoSrc  = cfg.slytherinVideo  ? `videos/${cfg.slytherinVideo}`  : null;
  HOUSES.ravenclaw.videoSrc  = cfg.ravenclawVideo  ? `videos/${cfg.ravenclawVideo}`  : null;
  HOUSES.hufflepuff.videoSrc = cfg.hufflepuffVideo ? `videos/${cfg.hufflepuffVideo}` : null;

  // Wire badge click listeners
  document.querySelectorAll('.house-badge').forEach(btn => {
    btn.addEventListener('click', () => selectHouse(btn.dataset.house));
  });

  // Activate default house without toast or spin (silent initial load)
  const defaultHouse = (cfg.defaultHouse || 'gryffindor').toLowerCase();
  selectHouse(defaultHouse, { showToast: false, spin: false });
}

// Bootstrap on DOM ready
document.addEventListener('DOMContentLoaded', () => {
  initHouseSelector(window.siteContent || {});
});
