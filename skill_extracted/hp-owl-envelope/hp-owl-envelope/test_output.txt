/**
 * HP Owl Envelope
 * - Syncs address name from siteContent.friendName
 * - Updates wax seal letter when house changes
 * - Triggers owl delivery burst on envelope tap
 *
 * Listens for CustomEvent 'houseChanged' (dispatched by hp-house-selector-ui)
 */

const HOUSE_INITIALS = {
  gryffindor: 'G',
  slytherin:  'S',
  ravenclaw:  'R',
  hufflepuff: 'H',
};

/**
 * Update the wax seal letter and CSS custom properties for the active house.
 * @param {string} houseName - lowercase house name
 * @param {string} [primary] - hex color (optional, from houseChanged event)
 * @param {string} [accent]  - hex color (optional, from houseChanged event)
 */
function updateSealForHouse(houseName, primary, accent) {
  const letterEl = document.getElementById('env-seal-letter');
  if (letterEl) {
    letterEl.textContent = HOUSE_INITIALS[houseName] || 'H';
  }

  // Update CSS variables so seal background + drip + letter color all react
  if (primary) document.documentElement.style.setProperty('--house-primary', primary);
  if (accent)  document.documentElement.style.setProperty('--house-accent', accent);
}

/**
 * Brief animation burst on owl-1, as if it just delivered the letter.
 * Restores normal speed after 800ms.
 */
function triggerOwlDelivery() {
  const deliveryOwl = document.querySelector('.owl-1');
  if (!deliveryOwl) return;

  deliveryOwl.classList.add('owl-delivering');
  setTimeout(() => deliveryOwl.classList.remove('owl-delivering'), 800);
}

// ── Listen for house changes from hp-house-selector-ui ──────
document.addEventListener('houseChanged', (e) => {
  const { house, primary, accent } = e.detail;
  updateSealForHouse(house, primary, accent);
});

// ── Owl delivery on envelope tap ────────────────────────────
const envelope = document.getElementById('envelope');
if (envelope) {
  envelope.addEventListener('click', triggerOwlDelivery);
}

// ── Initialise on DOM ready ──────────────────────────────────
document.addEventListener('DOMContentLoaded', () => {
  const cfg = window.siteContent || {};

  // Set address name from content.js
  const nameEl = document.querySelector('.env-address-name');
  if (nameEl && cfg.friendName) {
    nameEl.textContent = cfg.friendName;
  }

  // Set initial seal letter from defaultHouse
  const defaultHouse = (cfg.defaultHouse || 'gryffindor').toLowerCase();
  // Primary/accent not passed here — :root CSS vars cover the default
  updateSealForHouse(defaultHouse);
});

═══════════════════════════════════════════════════════════
Integration Notes
═══════════════════════════════════════════════════════════

HTML
  Replace the ENTIRE contents of #envelope-scene (or the whole div itself)
  with the HTML block above.
  Remove all 8 .env-particle divs — the 4 .env-owl divs replace them.

CSS
  Paste into your main stylesheet, or save as owl-envelope.css and link in <head>:
    <link rel="stylesheet" href="owl-envelope.css">
  Also add these two Google Fonts to <head> if not already present:
    <link href="https://fonts.googleapis.com/css2?family=Dancing+Script:wght@700&family=Cinzel&display=swap" rel="stylesheet">

JavaScript
  Save as owl-envelope.js and link before </body>:
    <script src="owl-envelope.js"></script>
  Load order: content.js → house-selector.js → owl-envelope.js

CSS variables
  The seal color uses --house-primary and --house-accent.
  If hp-house-selector-ui is installed, it sets these automatically on houseChanged.
  If not, the :root defaults in the CSS block (Gryffindor red/gold) apply.

Envelope sizing
  The owls use left: -30px → left: calc(100% + 40px) to drift across.
  This requires #envelope to have position: relative (add if missing).
  If #envelope has overflow: visible and you see owls outside the envelope,
  add overflow: hidden to #envelope.

Delivery animation
  The owl-1 burst triggers on every envelope tap.
  If your site only allows the envelope to open once, add { once: true }
  to the addEventListener call in owl-envelope.js.
*/
